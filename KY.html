Map of KY

library(ggplot2)
## Warning: package 'ggplot2' was built under R version 3.3.2
library(maps)
ky <- read.table("http://uncg.rstats.ninja/datasets/spatial/KY_cities.txt", header=T, sep=",")
ky_map <- map(database="state", region="Kentucky", plot=F)
kygg <- data.frame(lon=ky_map$x, lat=ky_map$y)
library(ggplot2)
ggplot(kygg, aes(lon, lat)) + 
  geom_path() + 
  geom_point(data=ky, aes(x=lon, y=lat, size=Population), alpha=0.8, color="blue") + 
  geom_text(data=dplyr::filter(ky, Population > 20000), aes(label=City)) + 
  theme_bw()
